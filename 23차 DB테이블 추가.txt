create table restaurant(
rest_no number primary key,
rest_name varchar2(200),
rest_loc varchar2(500),
man_no number references manager(man_no)
);

CREATE TABLE KEYWORD_NAME(
KEY_NAME VARCHAR2(500) PRIMARY KEY,
KEY_NAME_HIT NUMBER,
CUST_NO NUMBER REFERENCES customer(cust_no)
);
CREATE TABLE KEYWORD_LOC(
KEY_LOC VARCHAR2(100) primary key,
KEY_LOC_HIT NUMBER,
CUST_NO NUMBER REFERENCES CUSTOMER(CUST_NO)
);

CREATE TABLE SNS(
SNS_NO NUMBER PRIMARY KEY,
SNS_CONTENT VARCHAR2(2000),
SNS_LIKE NUMBER,
SNS_WRITE_DATE DATE, 
CUST_NO NUMBER REFERENCES CUSTOMER(CUST_NO),
PROVIDER_NO NUMBER REFERENCES PROVIDER(PROVIDER_NO)
);
CREATE TABLE SNS_IMG(
SNS_IMG_NO NUMBER PRIMARY KEY,
SNS_IMG_NAME VARCHAR2(200),
SNS_NO NUMBER REFERENCES SNS(SNS_NO)
);
CREATE TABLE SNS_REPLY(
REPLY_NO NUMBER PRIMARY KEY,
REPLY_CONTENT VARCHAR2(500),
SNS_NO NUMBER REFERENCES SNS(SNS_NO),
CUST_NO NUMBER REFERENCES CUSTOMER(CUST_NO),
PROVIDER_NO NUMBER REFERENCES PROVIDER(PROVIDER_NO)
);

alter table PROVIDER add primary key (PROVIDER_NO);
